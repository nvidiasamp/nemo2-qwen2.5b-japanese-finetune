# scripts/03_run_continual_learning.py
import nemo_run as run
from nemo.collections import llm

def setup_continual_training():
    recipe = llm.qwen25_500m.pretrain_recipe(
        dir="./models/checkpoints/qwen25_continual",
        name="japanese_continual_learning"
    )
    recipe.model.restore_from_path = './models/qwen25_0.5b.nemo' # 从导入的模型开始
    recipe.data = run.Config(
        llm.PreTrainingDataModule,
        paths=['./data/llm_jp_wiki/nemo_binary/ja_wiki_text_document'],
        seq_length=2048,
        micro_batch_size=4,
        global_batch_size=64
    )
    recipe.trainer.max_steps = 1000
    return recipe

if __name__ == "__main__":
    recipe = setup_continual_training()
